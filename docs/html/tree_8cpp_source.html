217static int tree_traversal_(Tree* tree, TreeNode** node, va_list* args, TreeNodeAction* actions[4],
218 size_t depth, TreeChildSide side) {
219 assert(tree);

364int tree_verify(Tree* tree, bool do_traversal) {
365 assert(tree);
366
367 int res = Tree::OK;
368
369 if (!tree_is_initialised(tree)) {
370 res |= Tree::UNITIALISED;
371 return res;
372 }
373
374 CHECK_ERR_(tree->elem_size <= 0, tree->INVALID_ELEM_T_SIZE);
375 CHECK_ERR_(tree->size < 0, tree->NEGATIVE_SIZE);
376
377 if (do_traversal) {
378 size_t cnt = 0;
379 res |= tree_preorder(tree, &tree->root, &tree_node_count_, &cnt);

382 }
